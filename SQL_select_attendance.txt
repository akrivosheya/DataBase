SELECT TOURISTS.NAME AS SPORTSMAN_NAME, TOURISTS.LAST_NAME AS SPORTSMAN_LAST_NAME, TOURISTS.BIRTH AS SPORTSMAN_BIRTH, TRAININGS.NAME AS TRAINING, ATTENDANCE.TIME AS TIME, ATTENDANCE.VISITED AS VISITED
   FROM (ATTENDANCE LEFT OUTER JOIN TOURISTS ON TOURISTS.ID = ATTENDANCE.SPORTSMAN)
      LEFT OUTER JOIN TRAININGS ON TRAININGS.ID = ATTENDANCE.TRAINING
