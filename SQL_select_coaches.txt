SELECT DISTINCT TOURISTS.NAME, TOURISTS.LAST_NAME, TOURISTS.SEX, TOURISTS.BIRTH, TOURISTS.CATEGORY, COACHES.SPECIALIZATION, SECTIONS.NAME, COACHES.SALARY
   FROM ((((COACHES LEFT OUTER JOIN TOURISTS ON TOURISTS.ID = COACHES.ID)
    LEFT OUTER JOIN SECTIONS ON COACHES.SECTION = SECTIONS.ID)
    LEFT OUTER JOIN TRAINS ON TRAINS.COACH = COACHES.ID)
    LEFT OUTER JOIN GROUPS ON TRAINS.GROUP_ID = GROUPS.ID)
    LEFT OUTER JOIN TRAININGS ON TRAINS.TRAINING = TRAININGS.ID
