SELECT TRAININGS.NAME, SUM(TRAININGS.ENDING_HOUR - TRAININGS.BEGINNING_HOUR) AS TIME
   FROM (((TRAININGS LEFT OUTER JOIN TRAINS  ON TRAINS.TRAINING = TRAININGS.ID )
      LEFT OUTER JOIN SECTIONS ON SECTIONS.ID = TRAININGS.SECTION)
      LEFT OUTER JOIN COACHES ON TRAINS.COACH = COACHES.ID)
      LEFT OUTER JOIN TOURISTS ON TOURISTS.ID = COACHES.ID
   GROUP BY TRAININGS.NAME
