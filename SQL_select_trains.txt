SELECT TOURISTS.NAME AS COACH_NAME, TOURISTS.LAST_NAME AS COACH_LAST_NAME, TOURISTS.BIRTH AS COACH_BIRTH, GROUPS.NAME AS GROUP_NAME, TRAININGS.NAME AS TRAINING, SECTIONS.NAME AS SECTION
   FROM TRAINS, TOURISTS, TRAININGS, GROUPS, SECTIONS
   WHERE TOURISTS.ID = TRAINS.COACH AND TRAININGS.ID = TRAINS.TRAINING AND GROUPS.ID = TRAINS.GROUP_ID AND SECTIONS.ID = TRAININGS.SECTION
