SELECT TOURISTS.NAME AS TOURIST_NAME, TOURISTS.LAST_NAME AS TOURIST_LAST_NAME, TOURISTS.BIRTH AS TOURIST_BIRTH, HIKE.NAME AS HIKE, CONDUCTED_HIKE.TIME AS TIME
   FROM WENT_TO_HIKE, TOURISTS, CONDUCTED_HIKE, HIKE
   WHERE TOURISTS.ID = WENT_TO_HIKE.TOURIST AND CONDUCTED_HIKE.ID = WENT_TO_HIKE.HIKE AND HIKE.ID = CONDUCTED_HIKE.HIKE
